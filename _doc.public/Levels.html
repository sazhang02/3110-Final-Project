Module Levels

module Levels: sig .. end

Representation of levels data

type t

The abstract type of values representing levels.

type pos = Board.coord

The type representing the position of a tile

type level_id = int

The type of level identifier. Each level's level_id is unique.

exception UnknownLevel of level_id

Raised when an unknown level is met.

exception InvalidTile of pos

Raised when an invalid tile is met.

val from_json : Yojson.Basic.t -> t

from_json j is the set of levels that j represents. Requires: j is a valid json representation.

val entrance_pipe : t -> level_id -> Board.tile

entrance_pipe levels lev a pipe with position pos of the entrance pipe of the level lev in levels levels.

val exit_pipe : t -> level_id -> Board.tile

exit_pipe levels lev is a tile with position pos of the exit pipe of the level lev in levels levels. Raises InvalidTile if the lev is the last level (i.e. There is no exit)

val next_level : t -> level_id -> level_id

next_level levels id is the level_id of the level to which the level lev exits. Raises UnkownLevel if lev is the last level.

val prev_level : t -> level_id -> level_id

prev_level levels is the level_id level lev enters. Raises UnknownLevel of level_id if lev is not a level in levels.

val coin_count : t -> level_id -> int

total_coin_count levels id is the sum of the number of coins on level number id and of the coin counts o n previous levels in levels

val is_final_level : t -> level_id -> bool

is_final levels level_id is true if level_id is the id of the last level in levels, else false

val final_level_id : t -> level_id

final_level_id levels is the id of the last level in levels

val make_board : t -> level_id -> Board.t

make_board levels id is a 16x16 board of Board.t type with information from level number id in levels.

val make_all_boards : t -> Board.t array

make_all_boards levels is a set of boards representing levels
